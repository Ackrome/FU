def prepare_var_data(R, F, p):
    """
      X  Y  VAR(p).
    R, F:  numpy-    ( T).
    p:  .
    
    :
    X:  (T - p) x (2p + 1)
    Y:  (T - p) x 2
    """
    T = len(R)
    #  X  Y
    #   ()  T - p
    #    X: (2p + 1) (R_lags + F_lags + )
    X = []
    Y = []
    
    for t in range(p, T):
        #   X_t
        row = []
        #   R
        for lag in range(1, p+1):
            row.append(R[t - lag])
        #   F
        for lag in range(1, p+1):
            row.append(F[t - lag])
        #  
        row.append(1.0)
        
        X.append(row)
        
        #   Y_t = (R_t, F_t)
        Y.append([R[t], F[t]])
    
    X = np.array(X)
    Y = np.array(Y)
    return X, Y

def fit_var_ols(X, Y):
    """
       A   Y = X A
     Y  (N x 2), X  (N x K), A   (K x 2).
     A, ,    .
    """
    # A_hat = (X^T X)^(-1) X^T Y
    # X^T X:  (K x K)
    # X^T Y:  (K x 2)
    
    #     (X^T X)
    XtX = X.T @ X
    XtY = X.T @ Y
    inv_XtX = np.linalg.inv(XtX)
    A_hat = inv_XtX @ XtY  #  (K, 2)
    
    #  E = Y - X A_hat
    E = Y - X @ A_hat  # (N x 2)
    
    #    
    N = X.shape[0]  #  
    Sigma_hat = (E.T @ E) / N  # (2 x 2)
    
    return A_hat, E, Sigma_hat, inv_XtX

def compute_bic(Sigma_hat, p, T_eff, num_params):
    """
     BIC   .
    
    Sigma_hat:    (2 x 2)
    p: 
    T_eff:    (T - p)
    num_params:    (k).
    """
    # BIC = (T_eff) * ln(det(Sigma_hat)) + num_params * ln(T_eff)
    det_Sigma = np.linalg.det(Sigma_hat)
    bic_val = T_eff * np.log(det_Sigma) + num_params * np.log(T_eff)
    return bic_val

def var_select_order(R, F, p_max):
    """
      p=1  p_max,   BIC.
     (p_opt,  (p, BIC(p))  ).
    """
    T = len(R)
    results = []
    for p in range(1, p_max+1):
        #  X  Y
        X, Y = prepare_var_data(R, F, p)
        T_eff = X.shape[0]  # T - p
        
        #  
        A_hat, E, Sigma_hat, inv_XtX = fit_var_ols(X, Y)
        
        #     (k = 2*(2p+1))
        num_params = 2*(2*p + 1)
        
        #  BIC
        bic_val = compute_bic(Sigma_hat, p, T_eff, num_params)
        
        results.append((p, bic_val))
    
    #  p   BIC
    results.sort(key=lambda x: x[1])  #   BIC
    p_opt = results[0][0]
    
    return p_opt, results

def var_forecast(R, F, A_hat, p, h):
    """
       h     .
    R, F -   ( T).
    A_hat -    ( (2p+1, 2)).
    p -  .
    h -    .

     : R_forecast, F_forecast (  h).
    """
    R_pred = list(R)  #   ,  
    F_pred = list(F)
    T_full = len(R_pred)

    for step in range(h):
        #   R_{T_full}  F_{T_full} ( )
        #  p  .
        X_new = []
        #  R
        for lag in range(1, p+1):
            X_new.append(R_pred[T_full - lag])
        #  F
        for lag in range(1, p+1):
            X_new.append(F_pred[T_full - lag])
        # 
        X_new.append(1.0)
        X_new = np.array(X_new)

        #  (R_hat, F_hat) = X_new @ A_hat
        # A_hat: (2p+1, 2)
        new_est = X_new @ A_hat  #  (2,) -> [R_hat, F_hat]

        R_pred.append(new_est[0])
        F_pred.append(new_est[1])
        T_full += 1

    #   h  
    R_forecast = R_pred[-h:]
    F_forecast = F_pred[-h:]
    return R_forecast, F_forecast
